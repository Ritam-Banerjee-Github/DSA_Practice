// K sorted means, k elemenets will always be sorted from index i(i.e, i+k) in the array

class Solution
{
    //Function to return the sorted array.
    ArrayList <Integer> nearlySorted(int arr[], int num, int k)
    {
        // your code here
        ArrayList<Integer> ans=new ArrayList<>();
        PriorityQueue<Integer> pq=new PriorityQueue<Integer>();
        int n=arr.length;
        for(int i=0;i<n;i++){
            pq.add(arr[i]);
            if(pq.size()>k){
                ans.add(pq.peek());
                pq.remove();
            }
        }
        while(!pq.isEmpty()){
            ans.add(pq.peek());
            pq.remove();
        }
        return ans;
    }
}

TC - O(NlogK)
SC - O(N)
