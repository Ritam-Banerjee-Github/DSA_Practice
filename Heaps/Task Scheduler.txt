import java.util.* ;
import java.io.*; 
public class Solution {
    public static int taskScheduler(String s, int n, int k) {

        // Write your code here.
        int[] freq=new int[26];
        for(int i=0;i<n;i++){
            freq[s.charAt(i)-'A']++;
        }
        Arrays.sort(freq);
       
        int idleTime=(freq[25]-1)*k;
        for(int i=24;i>=0;i--){
              idleTime-=Math.min(freq[25]-1,freq[i]);
        }
        return s.length()+Math.max(0,idleTime);
    }

}
