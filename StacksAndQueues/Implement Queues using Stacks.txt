class MyQueue {
    Stack<Integer> first=new Stack<>();
    Stack<Integer> second=new Stack<>();

    public MyQueue() {
        
    }
    
    public void push(int x) {
        first.push(x);
    }
    
    public int pop() {
        if(!second.isEmpty()){
            int ele=second.pop();
            return ele;
        }else{
            while(!first.isEmpty()){
                second.push(first.pop());
            }
            return second.pop();
        }
    }
    
    public int peek() {
         if(!second.isEmpty()){
            return second.peek();
        }else{
            while(!first.isEmpty()){
                second.push(first.pop());
            }
            return second.peek();
        }
    }
    
    public boolean empty() {
        return first.isEmpty() && second.isEmpty();
    }
}
