class Tree {
    // Function to find the minimum element in the given BST.
    int minValue(Node node) {
        if(node == null) return -1;
        if(node.left==null && node.right!=null) return node.data;
        if(node.left==null && node.right==null) return node.data;

        int low = minValue(node.left);

        
        return low;
    }
}
