public class Solution
{
    //Function to check whether a Binary Tree is BST or not.
    boolean isBST(Node root)
    {
        return isBSTUtility(root,Integer.MIN_VALUE,Integer.MAX_VALUE);
    }
    public boolean isBSTUtility(Node root, int min, int max){
        if(root==null)return true;
        if(root.data>=max || root.data<=min) return false;  

// >= || <= because as per question BST cannot comtain duplicate value. So we need to check for equality as well.

        return (isBSTUtility(root.left,min,root.data)&&isBSTUtility(root.right,root.data,max));
    }
}
