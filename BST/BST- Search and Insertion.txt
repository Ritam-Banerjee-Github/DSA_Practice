class Solution {
    // Function to insert a node in a BST.
    Node insert(Node root, int Key) {
        // your code here
        if(root==null){
          Node node = new Node(Key);
          return node;
        } 
        Node curr = root;
        Node ans;

        while(true){
            if(curr.data==Key){
                break;
            }
            else
            if(curr.data<Key){
                if(curr.right!=null){
                    curr=curr.right;
                }
                else{
                    curr.right=new Node(Key);
                    break;
                }
            }
            else
            if(curr.data>Key){
                if(curr.left!=null){
                    curr=curr.left;
                }
                else{
                    curr.left=new Node(Key);
                    break;
                }
            }
        }

        return root;
        
    }
}
