//If anytime, it reaches maximum element, then this will always be the winner. 
If k>arr.length, the whole array elements will keep on rotating from i+1 to n. But the winner will always be maximum element.
(Check k max val->(10^9)) 

class Solution {
    public int getWinner(int[] arr, int k) {
         int maxi=0;
         int n=arr.length;

         for(int i=0;i<n;i++){
             maxi=Math.max(maxi,arr[i]);
         }

         if(k>=n){
             return maxi;
         }

         int winner=arr[0];
         int winCount=0;

         for(int i=1;i<n;i++){
             if(arr[i]>winner){
                 winner=arr[i];
                 winCount=1;
             }else{
                 winCount++;
             }

             if(winCount==k || winner==maxi){
                 return winner;
             }
         }

         return -1;
    }
}
