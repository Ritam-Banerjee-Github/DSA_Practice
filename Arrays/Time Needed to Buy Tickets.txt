class Solution {
    public int timeRequiredToBuy(int[] tickets, int k) {
        int n=tickets.length;
        
        int ans=0;
        
        while(true){
            for(int i=0;i<n;i++){
                if(tickets[i]>0){
                    ans++;
                    tickets[i]--;
                }
                if(tickets[k]==0) return ans;
            }  
        }  
      
    }
}
