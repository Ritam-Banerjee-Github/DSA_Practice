Brute - Store all non zero element in temp list and then replace the array by iteration in the temp list;

Optimal -
class Solution {
    void pushZerosToEnd(int[] arr, int n) {
        // code here
        int j=-1;
        for(int i=0;i<n;i++){
            if(arr[i]==0){
                j=i;
                break;
            }
        }

        if(j==-1) return;

        for(int i=j+1;i<n;i++){
            if(arr[i]!=0){
                int temp=arr[i];    // Basic Swapping of arr[i] and arr[j]. j should always point to zero and i to non zero element
                arr[i]=0;
                arr[j]=temp;
                
                j++;
            }
        }
    }
}

Time complexity: O(n)

Space complexity: O(1)
