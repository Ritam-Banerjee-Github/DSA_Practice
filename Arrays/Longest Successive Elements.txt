//Note : In this sequence if values are same, then they wont break the current sequence count.


import java.util.*;

public class Solution {
    public static int longestSuccessiveElements(int []a) {
        // Write your code here.
        int ans=0;
        int last_small=Integer.MIN_VALUE;
        int longest=0;
        Arrays.sort(a);
        int n=a.length;
    
        for(int i=0;i<n;i++){
             if(a[i]-1==last_small){
                longest++;
                last_small=a[i];
                

             }else if(a[i]!=last_small){
                 longest=1;
                 last_small=a[i];
             }

             ans=Math.max(ans,longest);
        }

        return ans;
    }
}
