class Solution {
    public int minStartValue(int[] nums) {
        int min=Integer.MAX_VALUE;
        
        int n=nums.length;
        int sum=0;
        
        for(int i=0;i<n;i++){
          sum+=nums[i];
          min=Math.min(min,sum);
        }
        
        if(min>0) return 1;
        
        return Math.abs(min)+1;
    }
}
