class Solution {
    int print2largest(int arr[], int n) {
        // code here
        int slargest=-1;
        int largest=arr[0];

        for(int i=1;i<n;i++){
            if(arr[i]>largest){
                slargest=largest;
                largest=arr[i];
            }
            else
            if(arr[i]<largest && arr[i]>slargest){
                slargest=arr[i];
            }
        }
        return slargest;
    }
}

Expected Time Complexity: O(N)
Expected Auxiliary Space: O(1)
