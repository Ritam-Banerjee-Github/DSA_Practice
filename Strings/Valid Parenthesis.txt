class Solution {
    public boolean isValid(String s) {
        Stack<Character> stack = new Stack<Character>();
        for(int i=0;i<s.length();i++){
            char ch=s.charAt(i);
            switch(ch) {
                case '(' :
                          stack.push('(');
                          break;
                case '{' :
                          stack.push('{');
                          break;
                case '[' :
                          stack.push('[');
                          break;                    
                case ')' :
                           if(stack.isEmpty() || stack.pop()!='(') return false;
                           break;  
                case '}' :
                           if(stack.isEmpty() || stack.pop()!='{') return false;
                           break;      
                case ']' :
                           if(stack.isEmpty() || stack.pop()!='[') return false;
                           break;                                     
            }
        }
        return stack.isEmpty();
    }
}

TC - O(N)
SC - O(N)
