class Solution {
    public String minRemoveToMakeValid(String s) {
        int n=s.length();
        Stack<Integer> stack=new Stack<>();
        int[] invalid=new int[n];
        
        for(int i=0;i<n;i++){
            if(s.charAt(i)=='('){
                stack.push(i);
            }else if(s.charAt(i)==')'){
                if(!stack.isEmpty()){
                    stack.pop();
                }else{
                    invalid[i]=1;
                }
            }
        }
        
        while(!stack.isEmpty()){
            int index=stack.peek();
            invalid[index]=1;
            stack.pop();
        }
        
        StringBuilder ans=new StringBuilder();
        
        for(int i=0;i<n;i++){
            if(s.charAt(i)=='(' || s.charAt(i)==')'){
                if(invalid[i]==1) continue;
            }
            
            ans.append(s.charAt(i));
        }
        
        return ans.toString();
        
        
    }
}

TC : O(N) + O(N) + O(N)
SC : O(N) + O(N)
