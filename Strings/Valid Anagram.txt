class Solution {
    public boolean isAnagram(String s, String t) {
    
     int l1=s.length();
     int l2=t.length();
     if(l1!=l2) return false;
      int[] count=new int[26];
      for(int i=0;i<l1;i++){
          count[s.charAt(i)-'a']++;  //s.charAt(i)-'a' will return the ASCII value of s.charAt(i)
      }
       for(int i=0;i<l2;i++){
          count[t.charAt(i)-'a']--;
          if(count[t.charAt(i)-'a']==-1) return false;
      }
      return true;
    }
}

TC - 2*O(N)
SC - O(1)
