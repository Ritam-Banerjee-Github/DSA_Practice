class Solution {
    public int minSteps(String s, String t) {
        int n=s.length();
        
        int[] charS=new int[26];
        int[] charT=new int[26];
        
        for(int i=0;i<n;i++){
            charS[s.charAt(i)-'a']++;
            charT[t.charAt(i)-'a']++;
        }
        
        int ans=0;
        
        for(int i=0;i<26;i++){
            if(charS[i]>charT[i]){
                ans+=charS[i]-charT[i];
            }
        }
        
        return ans;
    }
}
