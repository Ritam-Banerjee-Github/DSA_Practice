//Observation: if len is 4, i.e "aaaa" thennumber of homogenous substring is (1+2+3+4)=10.
Sum of first n(in this example, here 4) natural numbers. Below logic wil take care of adding all numbers(from 1 to n, and then again from 1) 

class Solution {
    int mod=(int)(1e9+7);

    public int countHomogenous(String s) {
        int n=s.length();

        int len=0;
        int result=0;

        for(int i=0;i<n;i++){
            if(i>0 && s.charAt(i)==s.charAt(i-1)){
                len++;
            }else{
                len=1;
            }

            result=(result+len)%mod;
        }

        return result;
    }
}
