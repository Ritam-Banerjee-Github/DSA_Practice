class Solution {
    private void dfsUtil(TreeNode root,List<String> ans,String temp){
        if(root==null) return;

        temp=temp+root.val;

        if(root.left==null && root.right==null){
            ans.add(temp);
            return;
        }

        temp=temp+"->";
        dfsUtil(root.left,ans,temp);
        dfsUtil(root.right,ans,temp);
    }
    public List<String> binaryTreePaths(TreeNode root) {
        List<String> ans=new ArrayList<String>();
        String temp="";
        dfsUtil(root,ans,temp);
        return ans;
    }
}

TC - O(N)
SC -O(N)
