class Solution {
    public int nthUglyNumber(int n) {
        int[] dp = new int[n+1];
        dp[1]=1;
        int i2=1;  //1 is the 1st and smallest ugly number
        int i3=1;
        int i5=1;
        for(int i=2;i<=n;i++){
           int i2UglyNum=dp[i2]*2;
           int i3UglyNum=dp[i3]*3;
           int i5UglyNum=dp[i5]*5;

           int minUglyNum=Math.min(i2UglyNum,Math.min(i3UglyNum,i5UglyNum));
           dp[i]=minUglyNum;

           if(minUglyNum==i2UglyNum){
              i2++;
           }
           if(minUglyNum==i3UglyNum){  //We use muliple if statements separately instead of else if because the min value might be same for more than one of them so we need to update all given prime factor indexes
               i3++;
           }
           if(minUglyNum==i5UglyNum){
              i5++;
           }
        }
        return dp[n];
    }
}
