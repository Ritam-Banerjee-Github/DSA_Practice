class Solution {
    int ans=0;
    private void solve(int i,int n,int currXor,int[] nums){
        if(i>=n){
           ans+=currXor;
           return;
        }
        
        solve(i+1,n,currXor,nums);
        solve(i+1,n,currXor^nums[i],nums);    
   
    }
    public int subsetXORSum(int[] nums) {
        int n=nums.length;
        
        solve(0,n,0,nums);
        return ans;
    }
}
