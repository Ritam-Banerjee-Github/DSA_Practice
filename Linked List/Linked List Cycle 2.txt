public class Solution {
    private ListNode startNode(ListNode slow, ListNode fast){
        while(slow!=fast){
            slow=slow.next;
            fast=fast.next;
        }
        
        return slow;
    }
    public ListNode detectCycle(ListNode head) {
        ListNode slow=head;
        ListNode fast=head;
        
        if(head==null || head.next==null) return null;
        
        while(fast!=null && fast.next!=null){
            slow=slow.next;
            fast=fast.next.next;
            
            if(slow==fast){
                slow=head;
                return startNode(slow,fast);
            }
        }
        
        return null;
    }
}
