class Solution {
    public ListNode mergeInBetween(ListNode list1, int a, int b, ListNode list2) {

       ListNode list2Head=list2;
       ListNode list2Tail=list2;

        while(list2!=null){
            list2Tail=list2;
            list2=list2.next;
        }
        
        int count=-1;
        
        ListNode temp=list1;
        ListNode startPrev=null;
        ListNode endNext=null;
        
        while(count!=b+1){
            count++;
            if(count==a-1){
                startPrev=temp;
            }
            
            if(count==b){
                endNext=temp;
            }
            
            temp=temp.next;
        }
        
        startPrev.next=list2Head;
        list2Tail.next=endNext.next;
        
        endNext.next=null;
        return list1;
    }
}
