class Solution {
    public int[] rearrangeArray(int[] nums) {
        int n=nums.length;
        int[] ans=new int[n];
        
        int positive=0; 
//Note : Since the array lenght is even and it says starting index 0 will have positive number, so we can see all even indexes will have even number and all odd indexes will have negativ number

        int negative=1;
        for(int i=0;i<n;i++){
            if(nums[i]>0 && positive<n){
                ans[positive]=nums[i];
                positive+=2;
            }else if(nums[i]<0 && negative<n){
                ans[negative]=nums[i];
                negative+=2;
            }
            
        }
        
        return ans; 
    
    }
}
