class Solution {
    public long countSubarrays(int[] nums, int minK, int maxK) {
        long ans=0;
        
        int minKPosition=-1;
        int maxKPosition=-1;
        int culpritIndex=-1;
        
        
        for(int i=0;i<nums.length;i++){
            if(nums[i]==minK){
                minKPosition=i;
            }
            
            if(nums[i]==maxK){
                maxKPosition=i;
            }
            
            if(nums[i]<minK || nums[i]>maxK) culpritIndex=i;
            
            
            int smaller=Math.min(maxKPosition,minKPosition);
            long temp=smaller-culpritIndex;
            
            ans+=temp>0?temp:0;
        }
        
        return ans;
    }
}
