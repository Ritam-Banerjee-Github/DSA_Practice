class Solution {
    public long maximumHappinessSum(int[] happiness, int k) {
        long ans=0;
        int n=happiness.length;
        
        Arrays.sort(happiness);
        int count=0;
        
        for(int i=n-1;i>n-1-k;i--){
            if(happiness[i]<count) break;
            
            int currHappiness=happiness[i]-count;
            ans+=currHappiness;
            count++;
        }
        
        return ans;
        
    }
}

TC : O(NlogN) + O(N)
