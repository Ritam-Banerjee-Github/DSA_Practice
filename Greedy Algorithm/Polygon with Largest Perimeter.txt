class Solution {
    public long largestPerimeter(int[] nums) {
        int n=nums.length;
        Arrays.sort(nums);
        long ans=0;
        long currSum=nums[0]+nums[1];
        
        for(int i=2;i<n;i++){
            if(nums[i]<currSum){
                ans=Math.max(ans,currSum+nums[i]);
            }
            
            currSum+=nums[i];
        }
        
        return ans==0?-1:ans;
    }
}
